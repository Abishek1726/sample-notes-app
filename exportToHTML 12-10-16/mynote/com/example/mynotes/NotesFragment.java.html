46       @Override 
47       public void onActivityCreated(Bundle savedInstanceState) { 
48           super.onActivityCreated(savedInstanceState); 
49    
50           rv = (RecyclerView) getView().findViewById(R.id.notes_recycler); 
51    
52               if(!NotesDatabaseHelper.isContextSet()) 
53                   NotesDatabaseHelper.setContext(getActivity()); 
54    
55               db = NotesDatabaseHelper.getInstance(); 
56    
57           multiSelector = new NoteCardAdapter.MultiSelector(); 
58           noteList = db.getAllNotes(); 
59           NoteCardAdapter adapter = new NoteCardAdapter(noteList, multiSelector); 
60           adapter.setCardClickedListener(this); 
61           //rv.setLayoutManager(new LinearLayoutManager(getActivity())); 
62           createLayout(); 
63           rv.setAdapter(adapter); 
64    
65       }

67       @Override 
68       public void onCardClick(Note note) { 
69           Intent intent = new Intent(getActivity(), ContentActivity.class); 
70           intent.putExtra(MainActivity.NOTE_KEY, note); 
71           startActivity(intent); 
72       } 
73    
74       @Override 
75       public void onCardLongClick() { 
76           MainActivity activity = (MainActivity) getActivity(); 
77           NoteCardAdapter.mActionModeCallback actionModeCallback = new NoteCardAdapter.mActionModeCallback(multiSelector); 
78           actionModeCallback.setListener(this); 
79           actionMode = activity.startSupportActionMode(actionModeCallback); 
80       } 
81    
82       @Override 
83       public void onDeselect() { 
84           if (actionMode != null) 
85               actionMode.finish(); 
86       } 
87    
88       @Override 
89       public void onDeletePressed() { 
90    
91           NoteCardAdapter adapter = (NoteCardAdapter) rv.getAdapter(); 
92           Log.d("b size", String.valueOf(rv.getAdapter().getItemCount())); 
93           for (int x : multiSelector.selectedSet) { 
94               db.deleteNote(adapter.getNoteList().get(x).getID()); 
95           } 
96           adapter.setNoteList(db.getAllNotes()); 
97           ////should optimize 
98           adapter.notifyDataSetChanged(); 
99    
100   
101      } 
102   
103      @Override 
104      public void onUpButtonPressed() { 
105   
106          for (int x : multiSelector.selectedSet) { 
107              NoteCardAdapter.ViewHolder v = (NoteCardAdapter.ViewHolder) rv.findViewHolderForAdapterPosition(x); 
108   
109              if (v != null) 
110                  v.cardView.setActivated(false); 
111          } 
112   
113      } 
114   
115   
116      @Override 
117      public void onResume() { 
118          super.onResume(); 
119          if (rv != null) { 
120              NoteCardAdapter adapter = (NoteCardAdapter) rv.getAdapter(); 
121              adapter.setNoteList(db.getAllNotes()); 
122              ///should optimize 
123              adapter.notifyDataSetChanged(); 
124          } 
125      }

127      private void createLayout() { 
128          if (LAYOUT_TYPE == LAYOUT_LIST) { 
129              rv.setLayoutManager(new LinearLayoutManager(getActivity())); 
130              return; 
131          } 
132          if (LAYOUT_TYPE == LAYOUT_STAG_GRID) { 
133              rv.setLayoutManager(new StaggeredGridLayoutManager(2, StaggeredGridLayoutManager.VERTICAL)); 
134              return; 
135          } 
136      } 
137   
138      public void setLayoutType(int ID) 
139      { 
140          LAYOUT_TYPE = ID; 
141      }
