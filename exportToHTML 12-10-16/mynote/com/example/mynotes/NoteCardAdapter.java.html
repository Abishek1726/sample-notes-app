46       @Override 
47       public NoteCardAdapter.ViewHolder onCreateViewHolder(ViewGroup parent, int viewType) { 
48           CardView cv = (CardView) LayoutInflater.from(parent.getContext()).inflate(R.layout.note_card_layout, parent, false); 
49    
50           return new ViewHolder(cv); 
51       }

53       @Override 
54       public void onViewAttachedToWindow(ViewHolder holder) { 
55           super.onViewAttachedToWindow(holder); 
56           if(!multiSelector.isSelectable()) 
57               holder.cardView.setActivated(false); 
58    
59       } 
60    
61       @Override 
62       public void onBindViewHolder(NoteCardAdapter.ViewHolder holder, final int position) { 
63    
64           CardView cv = holder.cardView; 
65           TextView tv1 = (TextView) cv.findViewById(R.id.note_card_title); 
66           tv1.setText(noteList.get(position).getTitle()); 
67    
68           TextView tv2 = (TextView) cv.findViewById(R.id.note_card_text); 
69           tv2.setText(noteList.get(position).getText()); 
70    
71    
72           cv.setOnClickListener(new View.OnClickListener() { 
73               @Override 
74               public void onClick(View v) { 
75                   if (multiSelector.isSelectable()) { 
76                       v.setActivated(!v.isActivated()); 
77                       multiSelector.selectAt(position, v.isActivated()); 
78    
79                       if (multiSelector.selectedSet.size() == 0) { 
80                           cardClickedListener.onDeselect(); 
81                           multiSelector.isCardSelectable = false; 
82    
83                       } 
84    
85                   } else 
86                       cardClickedListener.onCardClick(noteList.get(position)); 
87               } 
88           }); 
89           cv.setOnLongClickListener(new View.OnLongClickListener() { 
90               @Override 
91               public boolean onLongClick(View v) { 
92                   if (!multiSelector.isSelectable()) { 
93                       v.setActivated(!v.isActivated()); 
94                       cardClickedListener.onCardLongClick(); 
95                       multiSelector.isCardSelectable = true; 
96                       multiSelector.selectAt(position, true); 
97                       return true; 
98                   } 
99                   return false; 
100              } 
101          }); 
102   
103          Log.d("bindview",String.valueOf(position)); 
104      } 
105   
106      @Override 
107      public int getItemCount() { 
108          return noteList.size(); 
109      }

136      static class mActionModeCallback implements ActionMode.Callback { 
137   
138          onActionModeClickListener listener; 
139          MultiSelector multiSelector; 
140   
141          public mActionModeCallback(MultiSelector multiSelector) { 
142              this.multiSelector = multiSelector; 
143          } 
144   
145          public void setListener(onActionModeClickListener listener) { 
146              this.listener = listener; 
147          } 
148   
149   
150          @Override 
151          public boolean onCreateActionMode(ActionMode mode, Menu menu) { 
152   
153              MenuInflater inflater = mode.getMenuInflater(); 
154              inflater.inflate(R.menu.appbar_context_menu, menu); 
155   
156   
157              return true; 
158          } 
159   
160          @Override 
161          public boolean onPrepareActionMode(ActionMode mode, Menu menu) { 
162              return false; 
163          } 
164   
165          @Override 
166          public boolean onActionItemClicked(ActionMode mode, MenuItem item) { 
167              switch (item.getItemId()) { 
168                  case R.id.delete_context_menu: { 
169   
170                      multiSelector.setCardSelectable(false); 
171                      listener.onDeletePressed(); 
172                      mode.finish(); 
173                      return true; 
174                  } 
175   
176              } 
177   
178              return false; 
179          } 
180   
181          @Override 
182          public void onDestroyActionMode(ActionMode mode) { 
183              if (multiSelector.isSelectable()) { 
184                  listener.onUpButtonPressed(); 
185                  multiSelector.setCardSelectable(false); 
186              } 
187              multiSelector.selectedSet.clear(); 
188          } 
189   
190      } 
191   
192      static class MultiSelector { 
193          HashSet<Integer> selectedSet; 
194          boolean isCardSelectable; 
195   
196          public MultiSelector() { 
197              this.selectedSet = new HashSet<Integer>(); 
198              this.isCardSelectable = false; 
199          } 
200   
201          public boolean isSelectable() { 
202              return isCardSelectable; 
203          } 
204   
205          public void setCardSelectable(boolean cardSelectable) { 
206              isCardSelectable = cardSelectable; 
207          } 
208   
209          public void selectAt(int x, boolean b) { 
210              if (!b) 
211                  selectedSet.remove(x); 
212              else 
213                  selectedSet.add(x); 
214          } 
215   
216          public boolean isSelected(int pos) { 
217              return selectedSet.contains(pos); 
218          } 
219      }
