37       @Override 
38       public View onCreateView(LayoutInflater inflater, ViewGroup container, 
39                                Bundle savedInstanceState) { 
40           // Inflate the layout for this fragment 
41           return inflater.inflate(R.layout.fragment_check_list_edit, container, false); 
42       } 
43    
44       @Override 
45       public void onActivityCreated(Bundle savedInstanceState) { 
46           super.onActivityCreated(savedInstanceState); 
47           newItemButton = (Button) getView().findViewById(R.id.new_checkview_button); 
48           saveButton = (ImageButton) getView().findViewById(R.id.save_cl); 
49           editText = (EditText)getView().findViewById(R.id.cl_title_et); 
50    
51           LinearLayout ll = (LinearLayout) getView().findViewById(R.id.ll1); 
52           checkList = new CheckList(getActivity(), ll); 
53           checkList.newCheckView(); 
54    
55           newItemButton.setOnClickListener(this); 
56           saveButton.setOnClickListener(this); 
57       } 
58    
59    
60       @Override 
61       public void onClick(View v) { 
62    
63           if (v.getId() == newItemButton.getId()) { 
64               checkList.newCheckView(); 
65           } else if (v.getId() == saveButton.getId()) { 
66               List<CheckList.CheckViewData> checkViewDataList = checkList.getCheckViewDataList(); 
67               Map<String, Boolean> cMap = new HashMap<>(); 
68               for (CheckList.CheckViewData cd : checkViewDataList) { 
69                   cMap.put(cd.getText(), cd.isChecked()); 
70               } 
71    
72               CheckNote checkNote = new CheckNote(editText.getText().toString(), cMap); 
73    
74               Set<Map.Entry<String,Boolean>> set = checkNote.getCheckNoteMap().entrySet(); 
75    
76               Iterator<Map.Entry<String,Boolean>> it = set.iterator(); 
77    
78               String toastStr = ""; 
79    
80               while(it.hasNext()) 
81               { 
82                   Map.Entry<String,Boolean> me = it.next(); 
83                   toastStr += me.getKey() +" "+me.getValue().toString()+"\n"; 
84               } 
85    
86               Toast.makeText(getActivity(),toastStr,Toast.LENGTH_LONG).show(); 
87               ///save in db 
88    
89               NotesDatabaseHelper.getInstance().insertCheckNote(checkNote); 
90    
91    
92           } 
93       } 
94   }
