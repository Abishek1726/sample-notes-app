109      public void insertCheckNote(CheckNote checkNote) 
110      { 
111          ContentValues checkNoteValues = new ContentValues(); 
112   
113          checkNoteValues.put(CheckNotesTable.COLUMN_TITLE,checkNote.getTitle()); 
114   
115          long id = getWritableDatabase().insert(CheckNotesTable.TABLE_NAME,null,checkNoteValues); 
116   
117          Set<Map.Entry<String,Boolean>> set = checkNote.getCheckNoteMap().entrySet(); 
118   
119          Iterator<Map.Entry<String,Boolean>> iterator = set.iterator(); 
120   
121          while (iterator.hasNext()) 
122          { 
123              ContentValues checkItemValues = new ContentValues(); 
124   
125              Map.Entry<String,Boolean> entry = iterator.next(); 
126   
127              checkItemValues.put(CheckItemTable.COLUMN_TEXT,entry.getKey()); 
128   
129              int checkFlag = (entry.getValue())?1:0; 
130   
131              checkItemValues.put(CheckItemTable.COLUMN_CHECKED,checkFlag); 
132              checkItemValues.put(CheckItemTable.COLUMN_CHECK_NOTES_ID_FOREIGN,id); 
133   
134              getWritableDatabase().insert(CheckItemTable.TABLE_NAME,null,checkItemValues); 
135          } 
136   
137          getWritableDatabase().close(); 
138   
139      }

141      public List<CheckNote> getCheckNoteList() 
142      { 
143          List<CheckNote> list = new ArrayList<>(); 
144   
145          Cursor c = getWritableDatabase().query(CheckNotesTable.TABLE_NAME,null,null,null,null,null,null); 
146   
147          String query = "SELECT *FROM "+CheckItemTable.TABLE_NAME+" INNER JOIN "+CheckNotesTable.TABLE_NAME+" ON "+ 
148                  CheckItemTable.TABLE_NAME+"."+CheckItemTable.COLUMN_CHECK_NOTES_ID_FOREIGN+" = "+ 
149                  CheckNotesTable.TABLE_NAME+"."+CheckNotesTable.COLUMN_PRIMARY+" WHERE "+CheckNotesTable.COLUMN_PRIMARY+" = ? ;"; 
150   
151          if(c.moveToFirst()) 
152          while (!c.isAfterLast()) 
153          { 
154              String title = c.getString(c.getColumnIndex(CheckNotesTable.COLUMN_TITLE)); 
155              long id = c.getLong(c.getColumnIndex(CheckNotesTable.COLUMN_PRIMARY)); 
156   
157              Cursor c2 = getWritableDatabase().rawQuery(query,new String[]{String.valueOf(id)}); 
158   
159              Map<String,Boolean> map = new HashMap<>(); 
160   
161              if(c2.moveToFirst()) 
162                  while(!c2.isAfterLast()) 
163                  { 
164                      String key = c2.getString(c2.getColumnIndex(CheckItemTable.COLUMN_TEXT)); 
165                      int v = c2.getInt(c2.getColumnIndex(CheckItemTable.COLUMN_CHECKED)); 
166                      boolean val = (v==1)?true:false; 
167   
168                      map.put(key,val); 
169   
170                      c2.moveToNext(); 
171                  } 
172   
173              list.add(new CheckNote(title,map)); 
174   
175              c.moveToNext(); 
176          } 
177   
178          getWritableDatabase().close(); 
179   
180          return list; 
181      }
