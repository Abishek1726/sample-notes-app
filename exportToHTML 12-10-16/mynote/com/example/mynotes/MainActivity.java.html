93       @Override 
94       public boolean onOptionsItemSelected(MenuItem item) { 
95    
96           if (drawerToggle.onOptionsItemSelected(item)) 
97               return true; 
98    
99           if (item.getItemId() == R.id.appbar_menu_layout_options) { 
100   
101              if (item.getTitle().toString().equals("list")) { 
102                  item.setIcon(R.mipmap.ic_grid_option); 
103                  item.setTitle("grid"); 
104                  currentLayoutType = "list"; 
105   
106              } else { 
107                  item.setIcon(R.mipmap.ic_list_option); 
108                  item.setTitle("list"); 
109                  currentLayoutType = "grid"; 
110   
111              } 
112              replaceFragment(selectedFragPosition); 
113          } 
114          return super.onOptionsItemSelected(item); 
115      }

124      private void replaceFragment(int pos) { 
125          selectedFragPosition = pos; 
126          int fragLayoutType; 
127   
128          ///Refactor later 
129          if (currentLayoutType.equals("list")) 
130              fragLayoutType = NotesFragment.LAYOUT_LIST; 
131          else 
132              fragLayoutType = NotesFragment.LAYOUT_STAG_GRID; 
133   
134          Fragment fragment = null; 
135          switch (selectedFragPosition) { 
136              case 0: { 
137                  fragment = new AllNotesFragment(); 
138              } 
139              break; 
140              case 1: { 
141                  fragment = new NotesFragment(); 
142                  NotesFragment fragment1 = (NotesFragment) fragment; 
143                  fragment1.setLayoutType(fragLayoutType); 
144              } 
145              break; 
146              case 2: { 
147                  fragment = new ChecklistFragment(); 
148              } 
149              break; 
150              case 3: { 
151                  fragment = new FavouritesFragment(); 
152              } 
153              break; 
154          } 
155   
156          if (fragment != null) 
157              getFragmentManager().beginTransaction().replace(R.id.nav_frame, fragment).commit(); 
158   
159   
160      }

162      @Override 
163      public void onClick(View v) { 
164   
165          switch (v.getId()) { 
166              case R.id.fab_main: { 
167                  if (!isFABclicked) { 
168                      fabNote.show(); 
169                      fabChecklist.show(); 
170                      isFABclicked = true; 
171                  } else { 
172                      fabNote.hide(); 
173                      fabChecklist.hide(); 
174                      isFABclicked = false; 
175                  } 
176              } 
177              break; 
178              case R.id.fab_new_text: { 
179                  isFABclicked = false; 
180                  fabNote.hide(); 
181                  fabChecklist.hide(); 
182   
183                  Intent intent = new Intent(this, ContentActivity.class); 
184                  intent.putExtra(FAB_TYPE,"new_note"); 
185                  startActivity(intent); 
186              } 
187              break; 
188              case R.id.fab_new_checklist: 
189              { 
190                  isFABclicked = false; 
191                  fabNote.hide(); 
192                  fabChecklist.hide(); 
193   
194                  Intent intent = new Intent(this, ContentActivity.class); 
195                  intent.putExtra(FAB_TYPE,"new_cl"); 
196                  startActivity(intent); 
197   
198              } 
199          } 
200      }
